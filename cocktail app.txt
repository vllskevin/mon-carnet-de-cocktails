    <h2>üé≤ Quiz Cocktail</h2>
    <button onclick="startQuiz()">Lancer un quiz</button>
    <div id="quizArea" style="margin-top:1rem;"></div>
  </div>

  <script>
    const form = document.getElementById('cocktailForm');
    const list = document.getElementById('cocktailList');
    const quizArea = document.getElementById('quizArea');

    function loadCocktails() {
      const cocktails = JSON.parse(localStorage.getItem('cocktails')) || [];
      list.innerHTML = '';
      cocktails.forEach((cocktail) => {
        const div = document.createElement('div');
        div.className = 'cocktail';
        div.innerHTML = `
          <h3>${cocktail.name}</h3>
          <p><strong>Ingr√©dients :</strong><br>${cocktail.ingredients.replace(/\n/g, '<br>')}</p>
          <p><strong>Instructions :</strong><br>${cocktail.instructions.replace(/\n/g, '<br>')}</p>
        `;
        list.appendChild(div);
      });
    }

    form.addEventListener('submit', (e) => {
      e.preventDefault();
      const name = document.getElementById('name').value.trim();
      const ingredients = document.getElementById('ingredients').value.trim();
      const instructions = document.getElementById('instructions').value.trim();

      if (!name || !ingredients || !instructions) return;

      const newCocktail = { name, ingredients, instructions };
      const cocktails = JSON.parse(localStorage.getItem('cocktails')) || [];
      cocktails.push(newCocktail);
      localStorage.setItem('cocktails', JSON.stringify(cocktails));

      form.reset();
      loadCocktails();
    });

    function startQuiz() {
      const cocktails = JSON.parse(localStorage.getItem('cocktails')) || [];
      if (cocktails.length === 0) {
        quizArea.innerHTML = '<p>Aucun cocktail enregistr√©.</p>';
        return;
      }
      const randomIndex = Math.floor(Math.random() * cocktails.length);
      const cocktail = cocktails[randomIndex];
      quizArea.innerHTML = `
        <h3>Quels sont les ingr√©dients du <strong>${cocktail.name}</strong> ?</h3>
        <textarea id="quizAnswer" rows="4" placeholder="√âcris ta r√©ponse ici..."></textarea>
        <button onclick="checkQuiz('${cocktail.ingredients.replace(/'/g, "\\'").replace(/\n/g, '\\n')}')">Voir la r√©ponse</button>
        <div id="quizFeedback" style="margin-top:1rem;"></div>
      `;
    }

    function checkQuiz(correctAnswer) {
      const userAnswer = document.getElementById('quizAnswer').value.trim();
      const feedback = document.getElementById('quizFeedback');
      feedback.innerHTML = `
        <p><strong>Ta r√©ponse :</strong><br>${userAnswer.replace(/\n/g, '<br>')}</p>
        <p><strong>Bonne r√©ponse :</strong><br>${correctAnswer.replace(/\\n/g, '<br>')}</p>
      `;
    }

    loadCocktails();
  </script>
